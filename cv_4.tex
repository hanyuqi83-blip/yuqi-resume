\begin{rSection}{Research Experience}

\begin{rSubsection}{Renewable Curtailment Mitigation and Port-Based Hydrogen Planning in Hauts-de-France (E4C Challenge)}{Dec 2025 -- Mar 2026}{}{}
\item Identified high-risk renewable curtailment periods in Hauts-de-France from a 30-minute wind and solar generation dataset, estimating an annual absorbable curtailment potential of about \textbf{0.81 TWh} with a peak of approximately \textbf{1958 MW}.
\item Developed a short-term \textbf{MILP} dispatch model at 30-minute resolution to optimize the coordinated absorption of curtailed electricity by \textbf{BESS}, \textbf{Power-to-Heat} (TES), and \textbf{Power-to-H\textsubscript{2}}; achieved absorption rates of \textbf{100.0\%}, \textbf{99.8\%}, and \textbf{93.1\%} under $\alpha=0.005$, $0.0066$, and $0.01$, respectively.
\item Developed a long-term annual \textbf{MILP} for port-based hydrogen infrastructure planning, obtaining an economically selected configuration of a \textbf{5.67 MW} electrolyzer and a \textbf{21.59 MW / 64.60 MWh} battery system, corresponding to \textbf{8434 MWh/year} of absorbed curtailed electricity.
\end{rSubsection}


\begin{rSubsection}{Techno-economic Assessment of Biomass-to-SAF Plant Repurposing}{Apr 2024 -- Sep 2024}{Research Intern}{\'Electricit\'e de France (EDF R\&D)}
\item[] \textit{Supervised by Bruno P\'echin\'e (EDF) and Prof.\ Dr.\ Beno\^it Gabrielle (Universit\'e Paris-Saclay).}
\item Assessed the feasibility of converting an existing biomass power plant into a \textbf{200{,}000 t/y SAF facility}, accounting for fixed site conditions and regional feedstock logistics.
\item Developed a combined \textbf{techno-economic analysis (TEA) and MILP optimisation model} to evaluate hydrogen sourcing options, transport costs, biomass supply regions, and process configuration under green H$_2$ price uncertainty.
\item Incorporated data from gasification technology suppliers to compare two production routes: \textbf{Gasification--FT} and \textbf{Gasification--MeOH--MTJ}, using modular process blocks with capacity and efficiency constraints.
\item Estimated \textbf{CAPEX, OPEX, NPV, IRR}, and break-even thresholds across multiple regional and hydrogen price scenarios; prepared internal technical documentation to support retrofit investment decisions.
\end{rSubsection}

\begin{rSubsection}{Numerical Simulation of Premixed H$_2$-Enriched Natural Gas Combustion}{Oct 2022 -- Jun 2023}{Undergraduate Thesis}{Shaanxi Provincial Key Laboratory of Thermal Science in Aero Power Systems}
